\begin{frame}{H1047R vs C420R vs R88Q (control)}
  \begin{itemize}
    \item 133,198 cells * 31,054 genes
    \item Values are raw integer UMI counts
    \item Roughly 54k cells from H1047R, 35k from C420R, 44k from R88Q
    \item QC filtered out low gene counts, high mitochondrial cells
    \item To correct sequencing depth differences, data was normalized to a target sum of 10k counts per cell
    \item Log-normalization was applied using log1p
    \item Aggregated individual cell counts into sample-level pseudobulk matrices across 10 biological replicates
    \item Differential expression was modeled using PyDESeq2 (Negative Binomial GLM)
    \item GSEA Pre-ranked was performed on PyDESeq2 Wald test statistics
  \end{itemize}
\end{frame}

\begin{frame}{Key takeaways}
  \begin{itemize}
    \item H1047R and C420R seem to attack the brain differently
    \item In H1047R, tumor cells upregulate synaptogenesis, cell adhesion, and synaptic transmission
    \item Neurons exposed to H1047R seem to suffer a metabolic collapse
    \item Excitotoxic over-firing triggers mitochondrial translation shutdown
    \item In C420R, tumor cells hijack muscle-like contractile networks
    \item Key tumor drivers include structural contractile machinery
    \item Neurons adapt to physical pressure via cytoskeletal remodeling
  \end{itemize}
\end{frame}

\begin{frame}{ANLS Mechanistic Reasoning \& Biological Interpretation}
  \begin{block}{Chain-of-Thought Breakdown of the ANLS Findings}
    \begin{itemize}
      \item \textbf{1. Hexokinase Isoform Switch}: Activity triggers a switch from baseline \emph{Hk1}/\emph{Hk3} to \emph{Hk2} ($0.145 \rightarrow 0.246$), driving aggressive glucose entry into GBM microenvironment cells.
      \item \textbf{2. Glycolytic Contracting vs. Funneling}: Downstream rate-limiting \emph{Pfkp} collapses ($33.4\% \rightarrow 4.8\%$), showing glucose carbon is funneled into lactate export or pentose phosphate pathways rather than complete intracellular glycolytic oxidation.
      \item \textbf{3. TCA Tuning \& Redox Switch}: Classical NAD+-dependent OxPhos (\emph{Cs}, \emph{Idh3a}, \emph{Sdhb}, \emph{Mdh2}) contracts to prevent mitochondrial damage. Crucially, mitochondrial \emph{Idh2} doubles ($0.185 \rightarrow 0.365$), producing protective NADPH against activity ROS.
      \item \textbf{4. Shuttle Program Confirmation}: \emph{Slc1a2} (glutamate import, 5.8x jump), \emph{Glul} (glutamine conversion, 3.3x jump), and \emph{Slc16a7} (MCT2 lactate import, 3.8x jump) prove that GBM cells DO activate an ANLS-like shuttle program upon stimulation.
    \end{itemize}
  \end{block}
\end{frame}
